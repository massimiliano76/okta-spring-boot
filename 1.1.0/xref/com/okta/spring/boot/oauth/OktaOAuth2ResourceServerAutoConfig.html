16  package com.okta.spring.boot.oauth;
17  
18  import com.okta.spring.boot.oauth.config.OktaOAuth2Properties;
19  import com.okta.spring.boot.oauth.http.UserAgentRequestInterceptor;
20  import org.springframework.boot.autoconfigure.AutoConfigureBefore;
21  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
22  import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
23  import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
24  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication;
25  import org.springframework.boot.autoconfigure.security.oauth2.resource.OAuth2ResourceServerProperties;
26  import org.springframework.boot.autoconfigure.security.oauth2.resource.servlet.OAuth2ResourceServerAutoConfiguration;
27  import org.springframework.boot.context.properties.EnableConfigurationProperties;
28  import org.springframework.context.annotation.Bean;
29  import org.springframework.context.annotation.Configuration;
30  import org.springframework.security.oauth2.jwt.JwtDecoder;
31  import org.springframework.security.oauth2.jwt.NimbusJwtDecoderJwkSupport;
32  import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationToken;
33  import org.springframework.web.client.RestOperations;
34  import org.springframework.web.client.RestTemplate;
35  
36  @Configuration
37  @AutoConfigureBefore(OAuth2ResourceServerAutoConfiguration.class)
38  @ConditionalOnClass(JwtAuthenticationToken.class)
39  @ConditionalOnProperty(name = "spring.security.oauth2.resourceserver.jwt.jwk-set-uri")
40  @EnableConfigurationProperties(OktaOAuth2Properties.class)
41  @ConditionalOnWebApplication(type = ConditionalOnWebApplication.Type.SERVLET)
42  class OktaOAuth2ResourceServerAutoConfig {
43  
44      @Bean
45      @ConditionalOnMissingBean
46      JwtDecoder jwtDecoder(OAuth2ResourceServerProperties oAuth2ResourceServerProperties,
47                            OktaOAuth2Properties oktaOAuth2Properties) {
48  
49              NimbusJwtDecoderJwkSupport decoder = new NimbusJwtDecoderJwkSupport(oAuth2ResourceServerProperties.getJwt().getJwkSetUri());
50              decoder.setJwtValidator(TokenUtil.jwtValidator(oAuth2ResourceServerProperties.getJwt().getIssuerUri(), oktaOAuth2Properties.getAudience()));
51              decoder.setRestOperations(restOperations());
52              return decoder;
53      }
54  
55      private RestOperations restOperations() {
56          RestTemplate restTemplate = new RestTemplate();
57          restTemplate.getInterceptors().add(new UserAgentRequestInterceptor());
58          return restTemplate;
59      }
60  }
